SELECT L.NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I JOIN ISSUESLISTS L ON I.ISSUESLIST_ID = L.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.FIRSTNAME, U.LASTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.FIRSTNAME, U.LASTNAME;
